思路：

- 对final计算前缀和数组pre。
- 每个查询[l,r]答案 = pre[r] - pre[l-1]。

代码参考：

```cpp
#include <iostream>
#include <vector>
using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, q;
    cin >> n >> m >> q;

    vector<ll> diff(n + 2, 0); // 差分数组，大小n+2防止越界

    for (int i = 0; i < m; ++i) {
        int l, r, c;
        cin >> l >> r >> c;
        diff[l] += c;
        diff[r + 1] -= c;
    }

    // 通过差分恢复原数组，并同时计算前缀和
    vector<ll> pre(n + 1, 0);
    ll cur = 0;
    for (int i = 1; i <= n; ++i) {
        cur += diff[i];
        pre[i] = pre[i - 1] + cur;
    }

    // 回答查询
    while (q--) {
        int l, r;
        cin >> l >> r;
        cout << pre[r] - pre[l - 1] << "\n";
    }

    return 0;
}
```

注意：这里pre[i]存储的是前i个元素的和（即a[1]到a[i]的和），与题1一致。

### 题5 参考答案

```cpp
#include <iostream>
#include <vector>

using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, q;
    cin >> n >> m >> q;

    vector<ll> diff(n + 2, 0);  // 差分数组，索引 1..n 有效，n+1 用于右边界减

    for (int i = 0; i < m; ++i) {
        int l, r, c;
        cin >> l >> r >> c;
        diff[l] += c;
        diff[r + 1] -= c;
    }

    vector<ll> pre(n + 1, 0);   // 前缀和，pre[i] 表示前 i 个元素的和
    ll cur = 0;
    for (int i = 1; i <= n; ++i) {
        cur += diff[i];          // 当前元素值
        pre[i] = pre[i - 1] + cur;
    }

    while (q--) {
        int l, r;
        cin >> l >> r;
        cout << pre[r] - pre[l - 1] << "\n";
    }

    return 0;
}
```

**算法说明**

1. 利用差分数组记录所有区间加操作（O(m)）。
2. 通过累加差分得到最终数组，同时计算前缀和（O(n)）。
3. 每个查询 O(1) 输出结果。
   总时间复杂度 O(n + m + q)，空间 O(n)。使用 `long long` 避免溢出。
